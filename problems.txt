var aBtn = document.getElementsByTagName('input');
for (var i=0; i<3; i++) {
	// 只有第三个函数有事件 innerHTML 会先清空页面上所有的按钮在创建
	document.body.innerHTML = document.body.innerHTML + '<input type="button" value="按钮" />';
	// 第一个按钮 = 没有事件处理函数了  0
	// 第二个按钮 = 没有事件处理函数了	 1
	// 第三个按钮 = 有事件处理函数了	 1
	aBtn[i].onclick = function () {
		alert(123);
	};
}
